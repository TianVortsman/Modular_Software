// Universal Document Modal - Enhanced Functionality
class UniversalDocumentModal {
    constructor() {
        this.state = {
            documentType: 'invoice',
            client: { type: 'business', name: '', email: '' },
            details: { docNumber: 'INV-2025-001', issueDate: '2025-08-02', dueDate: '2025-09-01' },
            lineItems: [
                { id: 1, type: 'standard', description: 'Web Design Services', detail: 'Initial wireframes and mockups for the new homepage.', qty: 25, price: 80.00, discount: '5%', vat: '15%' },
                { id: 2, type: 'vehicle', description: '2023 Toyota Hilux 2.8 GD-6', vin: 'VIN123456789', regNo: 'GP 987 654', qty: 1, price: 650000.00, discount: '0', vat: '15%' }
            ],
            summary: { subtotal: 651900.00, discount: 100.00, vat: 97770.00, total: 749570.00, paid: 0.00, balanceDue: 749570.00 },
            isSaving: false
        };

        this.documentConfig = {
            invoice: {
                title: 'Invoice',
                prefix: 'INV-',
                status: 'draft',
                fields: ['issueDate', 'dueDate'],
                actions: ['save', 'finalize', 'download', 'print', 'email']
            },
            quotation: {
                title: 'Quotation', 
                prefix: 'QUO-',
                status: 'draft',
                fields: ['issueDate', 'expiryDate'],
                actions: ['save', 'convert', 'download', 'print', 'email']
            },
            credit_note: {
                title: 'Credit Note',
                prefix: 'CRN-',
                status: 'draft', 
                fields: ['issueDate', 'originalRef'],
                actions: ['save', 'apply', 'download', 'print', 'email']
            },
            refund: {
                title: 'Refund',
                prefix: 'REF-',
                status: 'pending',
                fields: ['issueDate', 'refundReason'],
                actions: ['save', 'process', 'download', 'print', 'email']
            },
            recurring: {
                title: 'Recurring Invoice',
                prefix: 'REC-',
                status: 'active',
                fields: ['issueDate', 'frequency', 'endDate'],
                actions: ['save', 'activate', 'download', 'print', 'email']
            },
            vehicle_quote: {
                title: 'Vehicle Quotation',
                prefix: 'VQ-',
                status: 'draft',
                fields: ['issueDate', 'expiryDate', 'deliveryDate'],
                actions: ['save', 'convert', 'download', 'print', 'email']
            }
        };

        this.init();
    }

    init() {
        this.bindEvents();
        this.updateUI();
        this.setupAutoSave();
        
        // Initialize Lucide icons
        if (typeof lucide !== 'undefined') {
            lucide.createIcons();
        }
    }

    bindEvents() {
        // Document type change
        const docTypeSelect = document.getElementById('docType');
        if (docTypeSelect) {
            docTypeSelect.addEventListener('change', (e) => {
                this.handleDocumentTypeChange(e.target.value);
            });
        }

        // Client type toggle
        const clientToggle = document.getElementById('clientTypeToggle');
        if (clientToggle) {
            clientToggle.addEventListener('change', (e) => {
                this.state.client.type = e.target.checked ? 'individual' : 'business';
                this.updateClientUI();
            });
        }

        // Theme toggle (add button for this)
        document.addEventListener('keydown', (e) => {
            if (e.ctrlKey && e.key === 'd') {
                e.preventDefault();
                this.toggleTheme();
            }
        });

        // Auto-save on input changes
        document.addEventListener('input', (e) => {
            if (e.target.matches('input, select, textarea')) {
                this.handleInputChange(e);
            }
        });
    }

    handleDocumentTypeChange(newType) {
        this.state.documentType = newType;
        const config = this.documentConfig[newType];
        
        // Update document number
        this.state.details.docNumber = this.generateDocNumber(config.prefix);
        
        // Update title and status
        document.getElementById('documentTitle').textContent = `New ${config.title}`;
        const statusBadge = document.getElementById('documentStatus');
        statusBadge.textContent = config.status.charAt(0).toUpperCase() + config.status.slice(1);
        statusBadge.className = `status-badge status-${config.status}`;
        
        // Update document type indicator
        const indicator = document.querySelector('.document-type-indicator');
        indicator.setAttribute('data-doc-type', newType);
        
        // Update convert button text
        const convertBtn = document.getElementById('convertToInvoice');
        if (convertBtn && newType !== 'invoice') {
            convertBtn.innerHTML = `<i data-lucide="arrow-right-left"></i> Convert to ${this.documentConfig.invoice.title}`;
        }
        
        // Update document number input
        document.getElementById('docNumber').value = this.state.details.docNumber;
        
        // Re-initialize icons
        if (typeof lucide !== 'undefined') {
            lucide.createIcons();
        }
        
        this.updateUI();
    }

    generateDocNumber(prefix) {
        const year = new Date().getFullYear();
        const random = Math.floor(Math.random() * 999) + 1;
        return `${prefix}${year}-${random.toString().padStart(3, '0')}`;
    }

    updateClientUI() {
        const label = document.querySelector('label[for="clientName"]');
        if (label) {
            label.textContent = this.state.client.type === 'business' ? 'Client / Company Name' : 'Client Name';
        }
    }

    handleInputChange(e) {
        // Trigger auto-save indicator
        this.showSavingIndicator();
        
        // Update state based on input
        const { id, value } = e.target;
        if (id in this.state.details) {
            this.state.details[id] = value;
        } else if (id in this.state.client) {
            this.state.client[id] = value;
        }
        
        // Recalculate if it's a line item change
        if (e.target.closest('.line-item-row')) {
            this.recalculateSummary();
        }
    }

    showSavingIndicator() {
        const indicator = document.getElementById('autosaveIndicator');
        if (indicator) {
            indicator.style.opacity = '1';
            setTimeout(() => {
                indicator.style.opacity = '0';
            }, 2000);
        }
    }

    setupAutoSave() {
        setInterval(() => {
            this.showSavingIndicator();
        }, 10000);
    }

    recalculateSummary() {
        // Simple calculation for demo
        const summaryTotal = document.getElementById('grandTotal');
        if (summaryTotal) {
            summaryTotal.textContent = `R${this.state.summary.total.toLocaleString('en-ZA', { minimumFractionDigits: 2 })}`;
        }
    }

    toggleTheme() {
        const body = document.body;
        if (body.classList.contains('theme-light')) {
            body.classList.remove('theme-light');
            body.classList.add('theme-dark');
        } else {
            body.classList.remove('theme-dark');
            body.classList.add('theme-light');
        }
    }

    toggleClientSidebar() {
        const sidebar = document.querySelector('.client-sidebar');
        const icon = document.querySelector('#toggle-client-sidebar i');
        
        if (sidebar) {
            sidebar.classList.toggle('collapsed');
            
            // Update icon based on state
            if (sidebar.classList.contains('collapsed')) {
                icon.setAttribute('data-lucide', 'panel-right');
            } else {
                icon.setAttribute('data-lucide', 'panel-left');
            }
            
            // Re-initialize lucide icons
            if (typeof lucide !== 'undefined') {
                lucide.createIcons();
            }
        }
    }

    updateUI() {
        // Update any UI elements that depend on current state
        this.updateClientUI();
        this.recalculateSummary();
    }

    init() {
        // Initialize event listeners
        const toggleBtn = document.getElementById('toggle-client-sidebar');
        if (toggleBtn) {
            toggleBtn.addEventListener('click', () => this.toggleClientSidebar());
        }
    }
}

document.addEventListener('DOMContentLoaded', () => {
    // Add theme toggle button to header
    const headerRight = document.querySelector('.header-right');
    if (headerRight) {
        const themeToggle = document.createElement('button');
        themeToggle.className = 'btn-icon-only theme-toggle';
        themeToggle.innerHTML = '<i data-lucide="moon"></i>';
        themeToggle.title = 'Toggle dark mode (Ctrl+D)';
        themeToggle.addEventListener('click', () => {
            modal.toggleTheme();
            const icon = themeToggle.querySelector('i');
            const isDark = document.body.classList.contains('theme-dark');
            icon.setAttribute('data-lucide', isDark ? 'sun' : 'moon');
            if (typeof lucide !== 'undefined') {
                lucide.createIcons();
            }
        });
        
        headerRight.insertBefore(themeToggle, headerRight.lastElementChild);
    }
    
    // Initialize preview toggle
    const togglePreview = document.getElementById('toggle-preview');
    if (togglePreview) {
        togglePreview.addEventListener('click', () => {
            const preview = document.getElementById('live-preview-modal');
            const icon = togglePreview.querySelector('i');
            
            if (preview.style.display === 'none') {
                preview.style.display = 'block';
                icon.setAttribute('data-lucide', 'eye-off');
            } else {
                preview.style.display = 'none';
                icon.setAttribute('data-lucide', 'eye');
            }
            
            if (typeof lucide !== 'undefined') {
                lucide.createIcons();
            }
        });
    }
    
    // Initialize the modal
    window.modal = new UniversalDocumentModal();
    
    // Initialize live preview updates
    initLivePreview();
    
    // Initialize Lucide icons
    if (typeof lucide !== 'undefined') {
        lucide.createIcons();
    }
});

// Live Preview Functions
function initLivePreview() {
    // Update preview when inputs change
    document.addEventListener('input', updatePreview);
    document.addEventListener('change', updatePreview);
    
    // Initial preview update
    updatePreview();
}

function updatePreview() {
    // Update document type and number
    const docType = document.getElementById('document-type')?.value || 'invoice';
    const docNumber = document.getElementById('document-number')?.value || 'INV-2025-001';
    const issueDate = document.getElementById('current-date')?.value || '2025-08-02';
    
    document.getElementById('preview-title').textContent = docType.toUpperCase().replace('-', ' ');
    document.getElementById('preview-number').textContent = docNumber;
    document.getElementById('preview-date').textContent = issueDate;
    
    // Update client info
    const clientName = document.getElementById('client-name')?.value || 'Client Name';
    const clientAddress = document.getElementById('client-address-1')?.value || 'Client Address';
    
    document.getElementById('preview-client-name').textContent = clientName;
    document.getElementById('preview-client-address').textContent = clientAddress;
    
    // Update totals from summary
    const subtotal = document.getElementById('subtotal')?.textContent || 'R0.00';
    const tax = document.getElementById('tax-total')?.textContent || 'R0.00';
    const total = document.getElementById('final-total')?.textContent || 'R0.00';
    
    document.getElementById('preview-subtotal').textContent = subtotal;
    document.getElementById('preview-tax').textContent = tax;
    document.getElementById('preview-total').textContent = total;
}
